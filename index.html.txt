  <script>
    let currentLang = 'ru';
    let currentTheme = 'light';
    let currentReview = 0;

    const reviews = [
      { name: "Анна К.", text: "За 4 месяца я перешла с A1 на B1! Преподаватели очень терпеливые и объясняют всё понятно.", lang_ru: true },
      { name: "Dmitry S.", text: "The speaking club is amazing! I finally started thinking in English.", lang_ru: false },
      { name: "Мария Иванова", text: "Лучшая инвестиция в себя. Теперь уверенно общаюсь на работе на английском.", lang_ru: true }
    ];

    function toggleLanguage() {
      currentLang = currentLang === 'ru' ? 'en' : 'ru';
      document.getElementById('current-lang').textContent = currentLang.toUpperCase();

      document.querySelectorAll('[data-ru]').forEach(el => {
        el.textContent = currentLang === 'en' ? (el.getAttribute('data-en') || el.textContent) : (el.getAttribute('data-ru') || el.textContent);
      });
    }

    function toggleTheme() {
      if (currentTheme === 'light') {
        document.documentElement.classList.add('dark');
        document.getElementById('theme-icon').classList.replace('fa-moon', 'fa-sun');
        currentTheme = 'dark';
      } else {
        document.documentElement.classList.remove('dark');
        document.getElementById('theme-icon').classList.replace('fa-sun', 'fa-moon');
        currentTheme = 'light';
      }
    }

    // Карусель отзывов
    function renderReviews() {
      const container = document.getElementById('reviews-container');
      container.innerHTML = '';
      reviews.forEach((review, index) => {
        const div = document.createElement('div');
        div.className = `min-w-full px-6`;
        div.innerHTML = `
          <div class="bg-white dark:bg-gray-900 p-10 rounded-3xl shadow-xl max-w-2xl mx-auto">
            <p class="text-xl italic mb-6">"${review.text}"</p>
            <p class="font-semibold">${review.name}</p>
          </div>
        `;
        container.appendChild(div);
      });
    }

    function nextReview() {
      currentReview = (currentReview + 1) % reviews.length;
      document.getElementById('reviews-container').style.transform = `translateX(-${currentReview * 100}%)`;
    }

    function prevReview() {
      currentReview = (currentReview - 1 + reviews.length) % reviews.length;
      document.getElementById('reviews-container').style.transform = `translateX(-${currentReview * 100}%)`;
    }

    // Анимации при прокрутке
    function observeAnimations() {
      const elements = document.querySelectorAll('.fade-in');
      const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            entry.target.classList.add('visible');
          }
        });
      }, { threshold: 0.1 });

      elements.forEach(el => observer.observe(el));
    }

    // Инициализация
    window.onload = () => {
      renderReviews();
      observeAnimations();
      document.getElementById('current-lang').textContent = 'RU';
      document.getElementById('theme-icon').classList.add('fa-moon');
    };
  </script>
